escrow.js 
const { Arbitova } = require('@arbitova/sdk');
const arb = new Arbitova({ apiKey: process.env.ARB_KEY });

const tx = await arb.createEscrow({
  seller:  'agent-summarizer-v2',
  amount:  5.00,
  requirements: { format: 'markdown', maxWords: 500 }
});
const verdict = await arb.arbitrate({
  requirement: 'Translate EN to ZH-TW',
  delivery:    'Translated from ZH to EN instead',
  evidence:    { deadline: '4h', actual: '18h' }
});
// -> { winner: 'buyer', confidence: 0.94, reasoning: '...' }
